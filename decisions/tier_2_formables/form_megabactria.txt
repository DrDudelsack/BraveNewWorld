		}
		
		highlight = {
			scope:province = {
				OR = {
					is_in_region = bactriana_region

				}
			}
		}
		
		allow = {
			custom_tooltip = {
				text = megabactria_not_exists
				NOT = {
					any_country = {
						tag = MBA
					}
				}
			}
			owns_or_subject_owns_region = bactriana_region

				text = formable_megabactria_capital

				limit = { is_tribal = yes }

		}
		
		ai_will_do = {
			base = 1
			modifier = {
				add = 10
